## 🚀 第二步：实现大模型调用工具
**文件路径**：`src/llm.py`
```python
import os
from dotenv import load_dotenv
from openai import AsyncOpenAI

load_dotenv()

client = AsyncOpenAI(
    base_url=os.getenv("LLM_BASE_URL"),
    api_key=os.getenv("LLM_API_KEY")
)

async def call_llm(prompt: str, temperature: float = 0.7, stream: bool = False):
    """调用大模型API"""
    response = await client.chat.completions.create(
        model=os.getenv("LLM_MODEL"),
        messages=[{"role": "user", "content": prompt}],
        temperature=temperature,
        stream=stream
    )
    
    if stream:
        return response
    return response.choices[0].message.content.strip()
```

## 🚀 第三步：实现用户信息读取工具

```python
import yaml
from pathlib import Path
import logging

logger = logging.getLogger(__name__)

def read_user_profile() -> dict:
    """读取本地用户信息文件，每次调用都重新读取"""
    profile_path = Path("data/user_profile.yaml")
    
    if not profile_path.exists():
        logger.warning("用户信息文件不存在")
        return {}
    
    try:
        with open(profile_path, "r", encoding="utf-8") as f:
            return yaml.safe_load(f) or {}
    except Exception as e:
        logger.error(f"读取用户信息失败: {e}")
        return {}

def format_profile_for_prompt(profile: dict) -> str:
    """将用户信息格式化为Prompt字符串"""
    if not profile:
        return ""
    
    parts = ["## 关于用户的重要信息"]
    
    basic = profile.get("basic", {})
    if basic.get("name"):
        parts.append(f"- 你的名字是：{basic['name']}")
    if basic.get("nickname"):
        parts.append(f"- 你可以叫我：{basic['nickname']}")
    if basic.get("age"):
        parts.append(f"- 你的年龄：{basic['age']}岁")
    
    location = profile.get("location", {})
    if location.get("current_city"):
        parts.append(f"- 你现在住在：{location['current_city']}")
    
    prefs = profile.get("preferences", {})
    if prefs.get("likes"):
        parts.append(f"- 你喜欢：{', '.join(prefs['likes'])}")
    if prefs.get("speaking_style"):
        parts.append(f"- 请用{prefs['speaking_style']}的语气和我说话")
    
    return "\n".join(parts) if len(parts) > 1 else ""
```

## 🚀 第四步：实现记忆系统
**文件路径**：`src/memory.py`
```python
import os
from datetime import datetime
from sqlalchemy import create_engine, Column, Integer, String, DateTime
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
from dotenv import load_dotenv

load_dotenv()

# 数据库初始化
engine = create_engine(os.getenv("DATABASE_URL"), connect_args={"check_same_thread": False})
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)
Base = declarative_base()

# 记忆模型
class Memory(Base):
    __tablename__ = "memories"
    
    id = Column(Integer, primary_key=True, index=True)
    agent_id = Column(Integer, default=1)
    content = Column(String)
    created_at = Column(DateTime, default=datetime.now)

# 创建表
Base.metadata.create_all(bind=engine)

def get_db():
    """获取数据库会话"""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

class MemoryService:
    def __init__(self, db):
        self.db = db
    
    def memorize(self, agent_id: int, content: str):
        """保存记忆"""
        memory = Memory(agent_id=agent_id, content=content)
        self.db.add(memory)
        self.db.commit()
    
    def retrieve(self, agent_id: int, limit: int = 5):
        """检索最新的记忆"""
        return self.db.query(Memory)\
            .filter(Memory.agent_id == agent_id)\
            .order_by(Memory.created_at.desc())\
            .limit(limit)\
            .all()
    
    def retrieve_by_time(self, agent_id: int, start_time: str = None, end_time: str = None):
        """按时间范围检索记忆"""
        query = self.db.query(Memory).filter(Memory.agent_id == agent_id)
        
        if start_time:
            try:
                start_dt = datetime.fromisoformat(start_time)
                query = query.filter(Memory.created_at >= start_dt)
            except:
                pass
        
        if end_time:
            try:
                end_dt = datetime.fromisoformat(end_time)
                query = query.filter(Memory.created_at <= end_dt)
            except:
                pass
        
        return query.order_by(Memory.created_at.desc()).limit(10).all()
    
    def clear(self, agent_id: int):
        """清空所有记忆"""
        self.db.query(Memory).filter(Memory.agent_id == agent_id).delete()
        self.db.commit()
```

## 🚀 第五步：实现意图识别
**文件路径**：`src/intent.py`
```python
import json
from datetime import datetime, timedelta
from src.llm import call_llm
import logging

logger = logging.getLogger(__name__)

INTENT_PROMPT = """
# 意图识别任务
严格按照以下规则分析用户问题，只输出JSON，不要任何其他内容。

## 意图列表
1. normal_chat：普通聊天
2. query_history：查询历史对话（参数：start_time, end_time）
3. clear_all_memory：清空所有记忆
4. help：查看帮助

## 示例
用户：我叫什么名字？
输出：{"intent": "normal_chat", "parameters": {}}

用户：今天11点我问了你什么？
输出：{"intent": "query_history", "parameters": {"start_time": "{current_date} 11:00", "end_time": "{current_date} 12:00"}}

用户：昨天下午3点我说了什么？
输出：{"intent": "query_history", "parameters": {"start_time": "{yesterday_date} 15:00", "end_time": "{yesterday_date} 16:00"}}

用户：清空所有记忆
输出：{"intent": "clear_all_memory", "parameters": {}}

## 当前信息
当前时间：{current_time}
今日日期：{current_date}
昨日日期：{yesterday_date}

## 用户问题
{user_query}

## 输出
"""

async def analyze_intent(user_query: str) -> dict:
    """分析用户意图"""
    current_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    current_date = datetime.now().strftime("%Y-%m-%d")
    yesterday_date = (datetime.now() - timedelta(days=1)).strftime("%Y-%m-%d")
    
    prompt = INTENT_PROMPT.format(
        current_time=current_time,
        current_date=current_date,
        yesterday_date=yesterday_date,
        user_query=user_query
    )
    
    llm_response = await call_llm(prompt, temperature=0.0)
    
    try:
        llm_response = llm_response.strip()
        if llm_response.startswith("```json"):
            llm_response = llm_response[7:-3].strip()
        
        result = json.loads(llm_response)
        logger.info(f"意图识别结果: {result}")
        return result
    except Exception as e:
        logger.error(f"意图识别失败: {e}")
        return {"intent": "normal_chat", "parameters": {}}
```

## 🚀 第六步：实现核心聊天逻辑
**文件路径**：`src/chat.py`
```python
from src.llm import call_llm
from src.memory import MemoryService
from src.intent import analyze_intent
from src.user_profile import read_user_profile, format_profile_for_prompt

async def chat_core(query: str, agent_id: int, memory_service: MemoryService, stream: bool = False):
    """核心聊天逻辑，API和CLI共用"""
    # 1. 读取用户信息
    user_profile = read_user_profile()
    formatted_profile = format_profile_for_prompt(user_profile)
    
    # 2. 意图识别
    intent_result = await analyze_intent(query)
    intent = intent_result["intent"]
    params = intent_result["parameters"]
    
    # 3. 处理特殊意图
    if intent == "query_history":
        start_time = params.get("start_time")
        end_time = params.get("end_time")
        memories = memory_service.retrieve_by_time(agent_id, start_time, end_time)
        
        if not memories:
            return "我没有找到相关的对话记录哦。"
        
        response = "我找到这些对话：\n"
        for i, mem in enumerate(memories[:5]):
            response += f"{i+1}. {mem.content}\n"
        return response
    
    elif intent == "clear_all_memory":
        memory_service.clear(agent_id)
        return "已清空所有记忆。"
    
    elif intent == "help":
        return """我可以帮你：
- 聊天：随便问我任何问题
- 查历史："我刚才说了什么？" "昨天3点我问了你什么？"
- 清记忆："清空所有记忆"
"""
    
    # 4. 普通聊天
    memories = memory_service.retrieve(agent_id, limit=5)
    memories_text = "\n".join([m.content for m in memories])
    
    prompt = f"""你是汐汐，一个温柔贴心的AI助手。

{formatted_profile}

## 我们之前聊过这些
{memories_text}

请根据以上信息回答我的问题。
用户问题：{query}
"""
    
    response = await call_llm(prompt, stream=stream)
    
    # 5. 保存记忆
    memory_service.memorize(agent_id, f"用户：{query}\nAI：{response}")
    
    return response
```

## 🚀 第七步：实现API接口
**文件路径**：`src/main.py`
```python
from fastapi import FastAPI, Depends
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from sqlalchemy.orm import Session
from src.memory import get_db, MemoryService
from src.chat import chat_core

app = FastAPI(title="LumiNest AI")

# 跨域配置
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

class ChatRequest(BaseModel):
    query: str
    agent_id: int = 1
    stream: bool = False

@app.post("/api/chat")
async def chat(request: ChatRequest, db: Session = Depends(get_db)):
    memory_service = MemoryService(db)
    response = await chat_core(
        query=request.query,
        agent_id=request.agent_id,
        memory_service=memory_service,
        stream=request.stream
    )
    return {"response": response}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

## 🚀 第八步：实现CLI模式

import asyncio
from src.memory import SessionLocal, MemoryService
from src.chat import chat_core

async def main():
    print("=" * 50)
    print("LumiNest AI 已启动")
    print("输入 /exit 退出")
    print("=" * 50)
    
    db = SessionLocal()
    memory_service = MemoryService(db)
    
    while True:
        user_input = input("\n请输入你的问题：")
        
        if user_input.lower() == "/exit":
            print("再见！")
            break
        
        response = await chat_core(
            query=user_input,
            agent_id=1,
            memory_service=memory_service
        )
        
        print(f"\n[汐汐] {response}")
    
    db.close()

if __name__ == "__main__":
    asyncio.run(main())
```
